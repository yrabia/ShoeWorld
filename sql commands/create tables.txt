CREATE TABLE INVENTORY
(
SHOE_ID				INTEGER NOT NULL,
SHOE_NAME				TEXT NOT NULL,
SHOE_COST				INTEGER NOT NULL,
SHOE_PRICE				INTEGER NOT NULL,
QUANTITY_ON_HAND		INTEGER NOT NULL,
QUANTITY_ON_ORDER		INTEGER NOT NULL,
QUANTITY_RECOMMENDED	INTEGER NOT NULL
);
ALTER TABLE INVENTORY
ADD CONSTRAINT INVENTORY_PK_SHOE_ID
PRIMARY KEY (SHOE_ID);

(
SHOE_ID				INTEGER NOT NULL,

);
ALTER TABLE SHOE_TYPES
ADD CONSTRAINT SHOE_TYPES_PK_SHOE_TYPES_ID

ALTER TABLE SHOE_TYPES
ADD CONSTRAINT SHOE_TYPES_FK_SHOE_ID
FOREIGN KEY (SHOE_ID)
REFERENCES INVENTORY (SHOE_ID);
